# Objetivo
# Predecir posibles canciones que un usuario podra seguir escuchando en el futuro,
# basndose en sus preferencias actuales y las caractersticas musicales de sus canciones favoritas.

# Filtrar canciones favoritas
liked_songs = df[df['liked'] == 1]

# Columnas a analizar
features = ['danceability', 'energy', 'valence', 'tempo', 'acousticness']

# Perfil promedio del usuario
user_profile = liked_songs[features].mean()

# Calcular similitud con todas las canciones
def similarity(row):
    return np.linalg.norm(user_profile - row[features])

df['similarity'] = df.apply(similarity, axis=1)

# Sugerir top 10 canciones ms parecidas que an no ha marcado como liked
recommendations = df[df['liked']==0].sort_values('similarity').head(10)

#Objetivo
#Identificar y sugerir posibles canciones que un usuario podra seguir escuchando el prximo ao,
#basndose en las caractersticas de sus canciones favoritas actuales y en la similitud musical.
import matplotlib.pyplot as plt
import seaborn as sns

# Ordenamos las recomendaciones por similitud
recommendations_sorted = recommendations.sort_values('similarity')

# Top 5 canciones recomendadas (usando el ndice como identificador)
top5 = recommendations_sorted.head(5)

# Mostrar ranking Top 5 en consola
print("Ranking Top 5 canciones recomendadas (por ndice):")
for idx, row in enumerate(top5.itertuples(), 1):
    print(f"{idx}. ndice {row.Index} - Danceability: {row.danceability}, Energy: {row.energy}, Valence: {row.valence}")

# Eje de tiempo: 12 meses
months = np.arange(1, 13)

plt.figure(figsize=(12,6))

# Colores para cada top
colors = ['red', 'blue', 'green', 'orange', 'purple']

# Graficamos todas las recomendaciones en gris
plt.scatter(recommendations_sorted['danceability'], recommendations_sorted['energy'], color='lightgray', label='Otras recomendaciones')

# Graficamos Top 5 con proyeccin de popularidad
for idx, row in enumerate(top5.itertuples()):
    # Simulacin de escucha creciente
    popularity = np.exp(-0.1*months) * (6 - idx)  # Top 1 mayor, Top 5 menor
    plt.plot(months, popularity, color=colors[idx], marker='o', label=f'Top {idx+1}: ndice {row.Index}')
    
    # Colocamos etiqueta en el ltimo punto
    plt.text(months[-1]+0.2, popularity[-1], f'Top {idx+1}', color=colors[idx])

plt.title('Proyeccin de escucha de las Top 5 canciones hacia el prximo ao')
plt.xlabel('Meses')
plt.ylabel('Nivel de escucha estimado')
plt.xticks(months)
plt.legend()
plt.show()

from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

# Seleccionamos canciones que le gustan actualmente
liked_songs = df[df['liked'] == 1].copy()

## Elegimos caracteristicas importantes
#Cada cancin tiene diferentes etiquetas que la describen:
#danceability  qu tanto te dan ganas de bailar.
#energy  qu tan energtica es.
#valence  qu tan feliz suena.

X = liked_songs[['danceability', 'energy', 'valence']]

# Normalizamos los datos
# Normalizar es como poner a todas las canciones en la misma escala, 
# para que ninguna domine a las dems por tener nmeros ms grandes.

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Definimos nmero de clusters (por ejemplo 3)

#Aqu usamos un algoritmo llamado KMeans, que es como jugar a hacer grupos de amigos:
#Le decimos que haga 3 grupos (n_clusters=3).
#Mira las canciones y las coloca en el grupo que ms se parezcan entre s 
#segn danceability, energy y valence.

kmeans = KMeans(n_clusters=3, random_state=42)
liked_songs['cluster'] = kmeans.fit_predict(X_scaled)

# Colores para cada cluster
colors = ['red', 'blue', 'green']

# Scatter plot de danceability vs valence con cluster
plt.figure(figsize=(10,6))
for cluster in range(3):
    cluster_data = liked_songs[liked_songs['cluster'] == cluster]
    plt.scatter(cluster_data['danceability'], cluster_data['valence'], 
                color=colors[cluster], label=f'Cluster {cluster}', alpha=0.6, s=100)

plt.title('Clustering de canciones favoritas segn danceability, energy y valence', fontsize=14)
plt.xlabel('Danceability')
plt.ylabel('Valence (felicidad)')
plt.legend()
plt.grid(True, linestyle='--', alpha=0.5)
plt.show()
